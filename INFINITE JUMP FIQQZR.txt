-- DRAG FUNCTIONALITY
local dragging = false
local dragInput
local dragStart
local startPos

local function updateDrag(input)
    local delta = input.Position - dragStart
    container.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end

titleBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = container.Position
        
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

titleBar.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        dragInput = input
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        updateDrag(input)
    end
end)

-- ==============================================
-- INFINITE JUMP FUNCTION
-- ==============================================
local function toggleInfiniteJump()
    infiniteJumpEnabled = not infiniteJumpEnabled
    
    statusValue.Text = infiniteJumpEnabled and "ON" or "OFF"
    statusValue.TextColor3 = infiniteJumpEnabled and Color3.fromRGB(0, 255, 0) or Color3.fromRGB(255, 0, 0)
    toggleBtn.Text = infiniteJumpEnabled and "ON" or "OFF"
    toggleBtn.BackgroundColor3 = infiniteJumpEnabled and Color3.fromRGB(0, 150, 0) or Color3.fromRGB(50, 50, 60)
    
    if infiniteJumpEnabled then
        jumpConnection = UserInputService.JumpRequest:Connect(function()
            local humanoid = player.Character and player.Character:FindFirstChildOfClass("Humanoid")
            if humanoid then
                humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
                humanoid.JumpPower = jumpHeight
            else
                player.CharacterAdded:Connect(function(char)
                    local newHumanoid = char:WaitForChild("Humanoid")
                    newHumanoid.JumpPower = jumpHeight
                end)
            end
        end)
    else
        if jumpConnection then
            jumpConnection:Disconnect()
            jumpConnection = nil
        end
        local humanoid = player.Character and player.Character:FindFirstChildOfClass("Humanoid")
        if humanoid then
            humanoid.JumpPower = 50
        end
    end
end

-- ==============================================
-- UPDATE JUMP HEIGHT
-- ==============================================
local function updateJumpHeight(newHeight)
    local num = tonumber(newHeight)
    if num then
        jumpHeight = math.clamp(math.floor(num), 10, 200)
        heightInput.Text = tostring(jumpHeight)
        
        if infiniteJumpEnabled then
            local humanoid = player.Character and player.Character:FindFirstChildOfClass("Humanoid")
            if humanoid then
                humanoid.JumpPower = jumpHeight
            end
        end
    else
        heightInput.Text = tostring(jumpHeight)
    end
end

heightInput.FocusLost:Connect(function()
    updateJumpHeight(heightInput.Text)
end)

toggleBtn.MouseButton1Click:Connect(toggleInfiniteJump)
