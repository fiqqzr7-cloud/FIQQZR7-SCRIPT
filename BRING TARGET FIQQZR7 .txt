-- ==============================================
-- FUNGSI DAPATKAN SEMUA PLAYER
-- ==============================================
local function getAllPlayers()
    local players = {}
    for _, plr in ipairs(Players:GetPlayers()) do
        if plr ~= player then
            table.insert(players, plr)
        end
    end
    return players
end

-- ==============================================
-- FUNGSI KIRA PARTS
-- ==============================================
local function countParts()
    local count = 0
    for _, v in ipairs(workspace:GetDescendants()) do
        if v:IsA("BasePart") and not v:IsDescendantOf(player.Character) and v.Parent ~= player.Character then
            if v.Name ~= "Baseplate" and v.Size.Magnitude < 50 then
                count = count + 1
            end
        end
    end
    return count
end

-- ==============================================
-- FUNGSI KIRA PLAYER BERDEKATAN
-- ==============================================
local function countNearbyPlayers()
    local count = 0
    local myPos = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
    
    if not myPos then return 0 end
    
    for _, plr in ipairs(Players:GetPlayers()) do
        if plr ~= player and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
            local hrp = plr.Character.HumanoidRootPart
            local dist = (hrp.Position - myPos.Position).Magnitude
            if dist <= config.radius then
                count = count + 1
            end
        end
    end
    return count
end

-- ==============================================
-- FUNGSI UPDATE SENARAI TARGET
-- ==============================================
local function updateTargetList()
    playerList = {}
    local myPos = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
    
    if not myPos then 
        task.wait(0.5)
        updateTargetList()
        return 
    end
    
    for _, plr in ipairs(Players:GetPlayers()) do
        if plr ~= player and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
            local hrp = plr.Character.HumanoidRootPart
            local dist = (hrp.Position - myPos.Position).Magnitude
            
            table.insert(playerList, {
                name = plr.Name,
                distance = math.floor(dist),
                player = plr,
                hrp = hrp
            })
        end
    end
    
    table.sort(playerList, function(a, b)
        return a.distance < b.distance
    end)
    
    refreshTargetList()
    updateCounts()
end

-- ==============================================
-- FUNGSI UPDATE COUNTS
-- ==============================================
local function updateCounts()
    partsFound = countParts()
    nearbyPlayers = countNearbyPlayers()
    
    if partsCountLabel then
        partsCountLabel.Text = tostring(partsFound)
    end
    if nearbyCountLabel then
        nearbyCountLabel.Text = tostring(nearbyPlayers)
    end
end

-- ==============================================
-- FUNGSI TARIK PARTS KE TARGET
-- ==============================================
local function pullPartsToTarget(target)
    if not target or not target.hrp then return 0 end
    
    local targetPos = target.hrp.Position
    local pulledCount = 0
    
    for _, v in ipairs(workspace:GetDescendants()) do
        if v:IsA("BasePart") and not v:IsDescendantOf(player.Character) then
            -- Abaikan baseplate dan part besar
            if v.Name ~= "Baseplate" and v.Size.Magnitude < 100 then
                local dist = (v.Position - targetPos).Magnitude
                if dist <= config.radius then
                    -- Buang bodyvelocity lama kalau ada
                    for _, child in pairs(v:GetChildren()) do
                        if child:IsA("BodyVelocity") then
                            child:Destroy()
                        end
                    end
                    
                    -- Create new BodyVelocity
                    local bv = Instance.new("BodyVelocity")
                    bv.Velocity = (targetPos - v.Position).Unit * config.power
                    bv.MaxForce = Vector3.new(9e4, 9e4, 9e4)
                    bv.P = 5000
                    bv.Parent = v
                    
                    -- Auto cleanup dalam 3 saat
                    task.delay(3, function()
                        if bv and bv.Parent then
                            bv:Destroy()
                        end
                    end)
                    
                    pulledCount = pulledCount + 1
                end
            end
        end
    end
    
    return pulledCount
end

-- ==============================================
-- FUNGSI MULA TARIK
-- ==============================================
local function startPulling()
    if not selectedTarget then
        notify("‚ùå Pilih target dulu!")
        return
    end
    
    if isPulling then
        stopPulling()
        return
    end
    
    isPulling = true
    updatePullButton()
    notify("üöÄ Menarik ke " .. selectedTarget.name)
    
    -- Loop untuk tarik berterusan
    pullConnection = RunService.Heartbeat:Connect(function()
        if not isPulling or not selectedTarget or not selectedTarget.hrp or not selectedTarget.hrp.Parent then
            stopPulling()
            return
        end
        
        local count = pullPartsToTarget(selectedTarget)
        if count > 0 then
            -- Update parts count maybe
            updateCounts()
        end
        
        task.wait(0.2)  -- Delay sikit supaya tak terlalu heavy
    end)
end

local function stopPulling()
    isPulling = false
    if pullConnection then
        pullConnection:Disconnect()
        pullConnection = nil
    end
    updatePullButton()
    notify("‚èπÔ∏è Berhenti")
end
