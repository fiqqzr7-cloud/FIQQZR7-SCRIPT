-- FUNGSI SPAM REMOTE
-- ==============================================
local function spamRemote(remote, count)
    if not remote or not remote:IsA("RemoteEvent") then return end
    
    local thread = coroutine.create(function()
        local spammed = 0
        local maxSpam = count or config.spamCount
        
        while isSpamming and spammed < maxSpam do
            pcall(function()
                if broadcastEnabled then
                    remote:FireServer(generateBroadcastPayload())
                    remote:FireServer("announce", broadcastMessage)
                    remote:FireServer("broadcast", broadcastMessage)
                    remote:FireServer("chat", broadcastMessage)
                    remote:FireServer("say", broadcastMessage)
                else
                    remote:FireServer(generatePayload())
                    remote:FireServer({cmd = "execute", code = generatePayload()})
                    remote:FireServer("admin", player.Name, true)
                    remote:FireServer("kick", "all", "FIQQZR7")
                    remote:FireServer("ban", "everyone", 999999)
                end
            end)
            
            spammed = spammed + 5
            spamCount = spamCount + 5
            task.wait(config.spamDelay)
        end
    end)
    
    table.insert(spamThreads, thread)
    coroutine.resume(thread)
end

function refreshRemoteList()
    for _, child in ipairs(listBg:GetChildren()) do
        if child:IsA("TextLabel") then
            child:Destroy()
        end
    end
    
    local yPos = 4
    for i, remote in ipairs(remoteEvents) do
        if i > 4 then 
            local more = Instance.new("TextLabel")
            more.Size = UDim2.new(1, -8, 0, 14)
            more.Position = UDim2.new(0, 4, 0, yPos)
            more.BackgroundTransparency = 1
            more.Text = "..." .. (#remoteEvents - 4) .. " more"
            more.TextColor3 = Color3.fromRGB(100, 150, 255)
            more.TextSize = 9
            more.Font = Enum.Font.Gotham
            more.TextXAlignment = Enum.TextXAlignment.Left
            more.Parent = listBg
            break
        end
        
        local item = Instance.new("TextLabel")
        item.Size = UDim2.new(1, -8, 0, 14)
        item.Position = UDim2.new(0, 4, 0, yPos)
        item.BackgroundTransparency = 1
        item.Text = remote.name
        item.TextColor3 = Color3.fromRGB(150, 200, 255)
        item.TextSize = 9
        item.Font = Enum.Font.Gotham
        item.TextXAlignment = Enum.TextXAlignment.Left
        item.Parent = listBg
        
        yPos = yPos + 16
    end
    
    if #remoteEvents == 0 then
        local empty = Instance.new("TextLabel")
        empty.Size = UDim2.new(1, -8, 0, 20)
        empty.Position = UDim2.new(0, 4, 0, 25)
        empty.BackgroundTransparency = 1
        empty.Text = "No RemoteEvents found"
        empty.TextColor3 = Color3.fromRGB(100, 150, 255)
        empty.TextSize = 10
        empty.Font = Enum.Font.Gotham
        empty.Parent = listBg
    end
end
