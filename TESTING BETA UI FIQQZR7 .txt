local function switchTab(tabName)
    for _, t in ipairs(tabs) do
        contentContainer[t .. "Btn"].BackgroundColor3 = Color3.fromRGB(22, 26, 36)
        contentContainer[t .. "Content"].Visible = false
    end
    contentContainer[tabName .. "Btn"].BackgroundColor3 = Color3.fromRGB(0, 180, 255)
    contentContainer[tabName .. "Content"].Visible = true
end

for _, t in ipairs(tabs) do
    contentContainer[t .. "Btn"].MouseButton1Click:Connect(function()
        switchTab(t)
    end)
end

local function addButton(parent, y, text, color, callback)
    local btn = Instance.new("TextButton")
    btn.Size = UDim2.new(0.9, 0, 0, 28)
    btn.Position = UDim2.new(0.05, 0, 0, y)
    btn.BackgroundColor3 = color or Color3.fromRGB(25, 30, 38)
    btn.TextColor3 = Color3.fromRGB(255, 255, 255)
    btn.Text = text
    btn.Font = Enum.Font.GothamBold
    btn.TextSize = 11
    btn.Parent = parent
    
    local btnCorner = Instance.new("UICorner")
    btnCorner.CornerRadius = UDim.new(0, 10)
    btnCorner.Parent = btn
    
    btn.MouseEnter:Connect(function()
        TweenService:Create(btn, TweenInfo.new(0.2), {
            BackgroundColor3 = Color3.fromRGB(0, 180, 255),
            Size = UDim2.new(0.92, 0, 0, 29)
        }):Play()
        btn.Position = UDim2.new(0.04, 0, 0, y-0.5)
    end)
    
    btn.MouseLeave:Connect(function()
        TweenService:Create(btn, TweenInfo.new(0.2), {
            BackgroundColor3 = color or Color3.fromRGB(25, 30, 38),
            Size = UDim2.new(0.9, 0, 0, 28)
        }):Play()
        btn.Position = UDim2.new(0.05, 0, 0, y)
    end)
    
    btn.MouseButton1Click:Connect(function()
        TweenService:Create(btn, TweenInfo.new(0.1), {BackgroundColor3 = Color3.fromRGB(0, 220, 255)}):Play()
        task.wait(0.1)
        TweenService:Create(btn, TweenInfo.new(0.1), {BackgroundColor3 = Color3.fromRGB(0, 180, 255)}):Play()
        task.wait(0.1)
        TweenService:Create(btn, TweenInfo.new(0.2), {BackgroundColor3 = color or Color3.fromRGB(25, 30, 38)}):Play()
        pcall(callback)
    end)
    
    return btn
end
